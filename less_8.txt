2. Не выполняя кода, ответить, что выведет браузер и почему:

if (!("a" in window)) {
var a = 1;
}
alert(a);

Будет undefined, т.к. по условию if строка должна быть неистинной, т.е. без значения, а она имеет числовое значение 1, значит условие не соблюдается.
---------------------------
var b = function a(x) {
x && a(--x);
};
alert(a);

Здесь не будет ничего, т.к. "а" - это имя функции и в теле функции не определено какое-либо действие.
----------------------------
function a(x) {
return x * 2;
}
var a;
alert(a);

Здесь не будет ничего, т.к. переменная "a" объявлена без указания значения, а значит в "alert" выводится функция "а", для которой значение переменной "x", указанной в функции, отсутствует.
----------------------------
function b(x, y, a) {
arguments[2] = 10;
alert(a);
}
b(1, 2, 3);

Здесь на экран веведется значение третьего аргумента, которому присвоено значение 10 в теле функции.
-----------------------------
function a() {
alert(this);
}
a.call(null);

Будет выведено, что это объект без названия.
